# напишите CMakeLists.txt для сборки трёх программ
# он должен иметь три цели: hmirror, vmirror и sobel

cmake_minimum_required(VERSION 3.11)

project(Negate CXX)
set(CMAKE_CXX_STANDARD 17)

# воспользуйтесь этой переменной при создании цели
set(IMGLIB_FILES img_lib.h img_lib.cpp
            ppm_image.h ppm_image.cpp)

set(HMIRROR_SRC  hmirr.cpp)
set(VMIRROR_SRC  vmirr.cpp)
set(SOBEL_SRC sobel.cpp)

if(CMAKE_SYSTEM_NAME MATCHES "^MINGW")
    set(SYSTEM_LIBS -lstdc++)
else()
    set(SYSTEM_LIBS)
endif()

add_executable(hmirror "${IMGLIB_FILES}" "${HMIRROR_SRC}")
add_executable(vmirror "${IMGLIB_FILES}" "${VMIRROR_SRC}")
add_executable(sobel "${IMGLIB_FILES}" "${SOBEL_SRC}")
